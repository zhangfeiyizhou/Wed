# 冒泡排序，两两比较，从小到大
<script>
  var a=[5,2,3,6,8,2,6,8,5,6]
  for (var i=0;i<a.length;i++){
    for(var j=0;j<a.length;j++){
      if(a[j]>a[j+1]){
        var timp=a[j]
            a[j]=a[j+1]
            a[j+1]=timp
      }
    }
  }
  console.log(a)
</script>
